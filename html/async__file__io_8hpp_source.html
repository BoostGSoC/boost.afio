599  template<bool isconst> struct void_type_selector { typedef void type; };
600  template<> struct void_type_selector<true> { typedef const void type; };
601  template<class T> struct async_file_io_dispatcher_rwconverter
602  {
603  typedef async_data_op_req<typename void_type_selector<std::is_const<T>::value>::type> return_type;
604  const std::vector<return_type> &operator()(const std::vector<async_data_op_req<T>> &ops)
605  {
606  typedef async_data_op_req<T> reqT;
607  static_assert(std::is_convertible<reqT, return_type>::value, "async_data_op_req<T> is not convertible to async_data_op_req<[const] void>");
608  static_assert(sizeof(return_type)==sizeof(reqT), "async_data_op_req<T> does not have the same size as async_data_op_req<[const] void>");
609  return reinterpret_cast<const std::vector<return_type> &>(ops);
610  }
611  };

614 inline async_io_op async_file_io_dispatcher_base::completion(const async_io_op &req, const std::pair<async_op_flags, std::function<async_file_io_dispatcher_base::completion_t>> &callback)
615 {
616  std::vector<async_io_op> r;
617  std::vector<std::pair<async_op_flags, std::function<async_file_io_dispatcher_base::completion_t>>> i;
618  r.reserve(1); i.reserve(1);
619  r.push_back(req);
620  i.push_back(callback);
621  return completion(r, i).front();
622 }
623 template<class R> inline async_io_op async_file_io_dispatcher_base::function(const async_io_op &req, std::function<R()> callback)
624 {
625  auto invoker=[](size_t, std::shared_ptr<detail::async_io_handle> _, std::function<R()> callback){
626  callback();
627  return std::make_pair(true, _);
628  };
629  return completion(req, std::make_pair(async_op_flags::None, std::bind(invoker, std::placeholders::_1, std::placeholders::_2, callback)));
630 }

632 {
633  auto invoker=[callback, args...](size_t, std::shared_ptr<detail::async_io_handle> _){
634  callback(args...);
635  return std::make_pair(true, _);
636  };
637  return completion(req, std::make_pair(async_op_flags::None, std::bind(invoker, std::placeholders::_1, std::placeholders::_2)));
